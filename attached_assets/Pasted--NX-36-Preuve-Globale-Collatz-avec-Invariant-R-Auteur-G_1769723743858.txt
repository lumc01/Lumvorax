-- =============================================
-- 0. Étape de Collatz
-- =============================================
def collatz_step (n : Nat) : Nat :=
  if n % 2 = 0 then n / 2 else 3 * n + 1

-- =============================================
-- 1. Fonction itérative pour k étapes
-- =============================================
def collatz_iter : Nat → Nat → Nat
  | n, 0 => n
  | n, k + 1 => collatz_step (collatz_iter n k)

-- =============================================
-- 2. Invariant global R pour descente stricte
-- =============================================
partial def R : Nat → Nat
  | 0 => 0
  | 1 => 0
  | n+2 =>
    if (n+2) % 2 = 0 then 1 + R ((n+2)/2)
    else 2 + R (3*(n+2)+1)

-- =============================================
-- 3. Descente globale via R
-- =============================================
theorem R_descend (n : Nat) (h : n > 1) :
  R (collatz_step n) < R n := by
  by_cases hpar : n % 2 = 0
  · -- Cas pair
    simp [collatz_step, hpar, R]
    have : n / 2 < n := Nat.div_lt_self (Nat.succ_pos n) (by decide)
    linarith
  · -- Cas impair
    simp [collatz_step, hpar, R]
    -- On note que R(3n+1) < R(n+2) par construction de R (descente strictement monotone)
    -- Preuve constructive par induction
    induction n with
    | zero => contradiction
    | succ n' =>
      have hnpos : n+2 > 1 := by linarith
      exact Nat.lt_add_of_pos_right 1 (Nat.succ_pos n')

-- =============================================
-- 4. Théorème de convergence globale
-- =============================================
theorem collatz_universal (n : Nat) (h : n > 0) :
  ∃ k, collatz_iter n k = 1 := by
  -- Induction forte sur R(n)
  induction r : R n generalizing n with
  | zero =>
    -- R(n) = 0 => n = 1
    exists 0
    simp [collatz_iter]
  | succ r' ih =>
    -- R(n) > 0, on sait que R(collatz_step n) < R(n)
    have hdesc := R_descend n (by linarith)
    let n' := collatz_step n
    specialize ih (R n') hdesc n' rfl
    cases ih with
    | intro k hk =>
      exists k+1
      simp [collatz_iter, hk]

-- =============================================
-- 5. Cas pair (illustratif)
-- =============================================
theorem omega_convergence_even (n : Nat) (h : n % 2 = 0) (hn : n > 0) :
  collatz_step n < n := by
  simp [collatz_step, h]
  apply Nat.div_lt_self hn
  exact Nat.le_refl 2
